git config --list[list user configuration]

git config --global user.name "name"

git config --global user.Email "user mailid"

git init [initialize git in cwf]

git add

git add -A[adding all changes to staging area]

git commit -m "commit msg"

git status [list all untrack files]

git push[local repo=>remote repo]

git clone [remote repo(exact copy)=> local workspace]

git pull[remote repo(change)=> local workspace]
